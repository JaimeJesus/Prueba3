CREATE TABLE Paciente(
    RUT INTEGER not null PRIMARY KEY,
    NOMBRE VARCHAR(30) NOT NULL,
    EDAD INTEGER not null,
    TELEFONO INTEGER not null,
    FECHA INTEGER NOT NULL,
    ID_TIPOENFERMEDAD INTEGER not null,
    ID_DOCTOR INTEGER NOT NULL,
    FOREIGN KEY (ID_TIPOENFERMEDAD) REFERENCES TIPOENFERMEDAD(ID_TIPOENFERMEDAD),
    FOREIGN KEY (ID_DOCTOR) REFERENCES DOCTOR(ID_DOCTOR)
);

CREATE TABLE Doctor(
    ID_DOCTOR INTEGER PRIMARY KEY,
    RUT INTEGER not null,
    NOMBRE VARCHAR(30) NOT NULL,
    ESPECIALIDAD varchar(30)
);



CREATE TABLE TIPOENFERMEDAD(
  ID_TIPOENFERMEDAD INTEGER PRIMARY KEY,
  TIPO VARCHAR(50) NOT NULL,
  COSTOTRATAMIENTO INTEGER
);
